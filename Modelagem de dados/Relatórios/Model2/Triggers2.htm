TR_Win32_CDROMDrive_AtivaComponente 
Name  TR_Win32_CDROMDrive_AtivaComponente 
Table  Win32_CDROMDrive 
Description   
SQL  CREATE TRIGGER TR_Win32_CDROMDrive_AtivaComponente ON Win32_CDROMDrive
FOR INSERT, UPDATE
AS
BEGIN
      UPDATE
           T
      SET
           CADC_Ativo = 1,
           CADC_DataDesativacao = NULL
      FROM
           Win32_CDROMDrive T
           JOIN
           INSERTED I ON I.SerialNumber_Win32_ComputerSystem = T.SerialNumber_Win32_ComputerSystem
           AND I.DeviceID = T.DeviceID
END 
TR_Win32_ComputerSystem_DataColeta 
Name  TR_Win32_ComputerSystem_DataColeta 
Table  Win32_ComputerSystem 
Description   
SQL  CREATE TRIGGER TR_Win32_ComputerSystem_DataColeta ON Win32_ComputerSystem
FOR INSERT, UPDATE
AS
      DECLARE @SN VARCHAR(255);
BEGIN
      SELECT @SN = SerialNumber_Win32_ComputerSystem FROM INSERTED;

      UPDATE Win32_ComputerSystem SET CADC_DataColeta = GETDATE()
      WHERE SerialNumber_Win32_ComputerSystem = @SN
END
