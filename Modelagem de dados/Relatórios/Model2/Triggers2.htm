TR_DATACOLETA 
Name  TR_DATACOLETA 
Table  Win32_ComputerSystem 
Description   
SQL  CREATE TRIGGER TR_DATACOLETA ON Win32_ComputerSystem
FOR INSERT
AS
      DECLARE @SerialNumber_Win32_ComputerSystem VARCHAR(255);
BEGIN
      SELECT @SerialNumber_Win32_ComputerSystem = SerialNumber_Win32_ComputerSystem FROM INSERTED;
      INSERT INTO Win32_ComputerSystem_CamposAdicionais (DataColeta, SerialNumber_Win32_ComputerSystem)
      VALUES (GETDATE(), @SerialNumber_Win32_ComputerSystem);
END
